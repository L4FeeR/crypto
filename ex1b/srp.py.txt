import math

class Solution:
    def is_prime(self, n: int) -> bool:
        if n < 2:
            return False
        for i in range(2, int(math.sqrt(n)) + 1):
            if n % i == 0:
                return False
        return True

    def isThree(self, n: int) -> bool:
        # Step 1: Find the square root
        root = int(math.sqrt(n))
        
        # Step 2: Check if n is a perfect square and the root is prime
        return root * root == n and self.is_prime(root)
